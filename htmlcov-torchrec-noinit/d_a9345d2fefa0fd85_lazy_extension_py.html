43def lazy_apply( 

44 module: torch.nn.Module, fn: Callable[[torch.nn.Module], None] 

45) -> torch.nn.Module: 

46 """Attaches a function to a module, which will be applied recursively to every 

47 submodule (as returned by `.children()`) of the module as well as the module itself 

48 right after the first forward pass (i.e. after all submodules and parameters have 

49 been initialized). 

50 

51 Typical use includes initializing the numerical value of the parameters of a lazy 

52 module (i.e. modules inherited from `LazyModuleMixin`). 

53 

54 NOTE: 

55 `lazy_apply()` can be used on both lazy and non-lazy modules. 

56 

57 Args: 

58 module (torch.nn.Module): module to recursively apply `fn` on. 

59 fn (Callable[[torch.nn.Module], None]): function to be attached to `module` and 

60 later be applied to each submodule of `module` and the `module` itself. 

61 

62 Returns: 

63 torch.nn.Module: `module` with `fn` attached. 

64 

65 Example:: 

66 

67 @torch.no_grad() 

68 def init_weights(m): 

69 print(m) 

70 if type(m) == torch.nn.LazyLinear: 

71 m.weight.fill_(1.0) 

72 print(m.weight) 

73 

74 linear = torch.nn.LazyLinear(2) 

75 lazy_apply(linear, init_weights) # doesn't run `init_weights` immediately 

76 input = torch.randn(2, 10) 

77 linear(input) # runs `init_weights` only once, right after first forward pass 

78 

79 seq = torch.nn.Sequential(torch.nn.LazyLinear(2), torch.nn.LazyLinear(2)) 

80 lazy_apply(seq, init_weights) # doesn't run `init_weights` immediately 

81 input = torch.randn(2, 10) 

82 seq(input) # runs `init_weights` only once, right after first forward pass 

83 """ 

84 

85 if not hasattr(module, "_functions_to_lazy_apply"): 

86 # pyre-ignore[16] 

87 module._functions_to_lazy_apply = [] 

88 if not hasattr(module, "_lazy_apply_hook"): 

89 # pyre-ignore[16] 

90 module._lazy_apply_hook = module.register_forward_hook( 

91 _apply_functions_after_first_forward 

92 ) 

93 # pyre-ignore[16] 

94 module._functions_to_lazy_apply.append(fn) 

95 return module

121 def apply(self, fn: Callable[[torch.nn.Module], None]) -> torch.nn.Module: 

122 """Applies `fn` recursively to every submodule (as returned by `.children()`) 

123 as well as self. Typical use includes initializing the parameters of a model. 

124 

125 NOTE: 

126 Calling `apply()` on an uninitialized lazy-module will result in an error. 

127 User is required to initialize a lazy-module (by doing a dummy forward pass) 

128 before calling `apply()` on the lazy-module. 

129 

130 Args: 

131 fn (torch.nn.Module -> None): function to be applied to each submodule. 

132 

133 Returns: 

134 torch.nn.Module: self 

135 

136 Example:: 

137 

138 @torch.no_grad() 

139 def init_weights(m): 

140 print(m) 

141 if type(m) == torch.nn.LazyLinear: 

142 m.weight.fill_(1.0) 

143 print(m.weight) 

144 

145 linear = torch.nn.LazyLinear(2) 

146 linear.apply(init_weights) # this fails, because `linear` (a lazy-module) hasn't been initialized yet 

147 

148 input = torch.randn(2, 10) 

149 linear(input) # run a dummy forward pass to initialize the lazy-module 

150 

151 linear.apply(init_weights) # this works now 

152 """ 

153 

154 if hasattr(self, "_initialize_hook"): 

155 raise RuntimeError( 

156 "Module {} has not been initialized. ".format(self) 

157 + "Please run a dummy forward pass on the model to initialize all modules, " 

158 + "or use torchrec.modules.lazy_extension.lazy_apply to attach a function " 

159 + "to this module which would be applied after this module is initialized." 

160 ) 

161 # If the module is already initialized, call `super().apply(fn)` to 

162 # run the usual apply logic. 

163 # pyre-ignore[16] 

164 return super().apply(fn)
